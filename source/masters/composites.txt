# Composites for BJCree project
# https://github.com/silnrsi/pysilfont/blob/master/docs/composite.md

### a-cans based
aa-cans         = a-cans + _dotabove@dot
wa-cans         = _dotleft & a-cans
waWestCree-cans = a-cans & _dotright
waa-cans        = wa-cans + _dotabove@dot
waaWestCree-cans = aa-cans & _dotright
waaNaskapi-cans = a-cans + _dotabove@dot2 + _dotabove@dot3
ayNaskapi-cans  = a-cans + _ring@ring
aayaNaskapi-cans = aa-cans + _ring@ringmiddle
wayNaskapi-cans = wa-cans + _ring@ring

### e-cans based
we-cans         = _dotleftclose & e-cans
weWestCree-cans = e-cans & _dotrightclose

### i-cans based
ii-cans         = i-cans + _dotabove@dot
wi-cans         = _dotleftclose & i-cans
wiWestCree-cans = i-cans & _dotrightclose
wii-cans        = wi-cans + _dotabove@dot
wiiWestCree-cans= ii-cans & _dotrightclose
aai-cans        = i-cans + _ring@ring

### o-cans based
oo-cans         = o-cans + _dotabove@dot
ooYCree-cans    = o-cans + _dot2hclose@2dotsmiddle
wo-cans         = _dotleft & o-cans
woWestCree-cans = o-cans & _dotright
woo-cans        = wo-cans + _dotabove@dot
wooWestCree-cans= oo-cans & _dotright
wooNaskapi-cans = o-cans + _dotabove@dot2 + _dotabove@dot3
oY-cans         = o-cans + _ring@ring
